public class Main {
    public static void main() {
        // Arrays
        Integer[] array = {0, 1, 2};
        for (Integer i : array) {
          System.out.println(i);
        }

        // Iterables
        java.util.ArrayList<Integer> iterable = new java.util.ArrayList<Integer>();
        iterable.add(0);
        iterable.add(1);
        iterable.add(2);
        for (Integer i : iterable) {
          System.out.println(i);
          System.out.println(2i);
          System.out.println(3i);
        }
    }
}
